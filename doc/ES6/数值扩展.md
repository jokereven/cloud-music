### `数值扩展`

#### `数值扩展`

```js
// E5S
// 十进制 -> 二进制
const a = 5; // 101
console.log(a.toString(2)); // 101

// 二进制 -> 十进制
const b = 101;
console.log(parseInt(b, 2)); // 5

// ES6 0B二进制  0O八进制
const a = 0b0101;
console.log(a);

const b = 0o101;
console.log(b);

// Number.isFinite()判断是不是一个有限的数
// 注意: 如果参数类型不是数值,返回的结果一律是false
console.log(Number.isFinite(520)); // true
console.log(Number.isFinite(Infinity)); // false
console.log(Number.isFinite(true)); // false

// NaN: 是不是不是一个数
console.log(Number.isNaN(NaN)); // true
console.log(Number.isNaN(520)); // false

console.log(Number.parseInt(3.14));
console.log(Number.parseFloat(3.14));
console.log(window.parseInt(3.14));
console.log(window.parseFloat(3.14));

// 整数
console.log(Number.isInteger(5));
console.log(Number.isInteger(5.5));

// IEEE 754 双精度标准

// 35 -> 00100011
// 0.375 -> 0.011
// 0.1 -> 0.000110011....
console.log(0.1000000000000001); // 0.1000000000000001
console.log(0.10000000000000001); // 0.1
console.log(0.10000000000000001 === 0.1); // true

// javascript的最大值
const max = Math.pow(2, 53);
console.log(max); // 2^53
console.log(max + 1); // 2^52
console.log(Number.MIN_SAFE_INTEGER); // 2^53-1 最大的安全数
console.log(Number.isSafeInteger(Number.MAX_SAFE_INTEGER)); // 判断是不是安全数

// Math.trunc 去除一个数的小数部分,返回整数部分
// 在操作这些非数值的时候,会先使用Number这个方法将其转换为数值
// 判断一个值,到底是正数,负数,还是0
console.log(Math.trunc(3.14)); // 3
console.log(Math.trunc(-3.14)); // -3
console.log(Math.trunc(true)); // 1
console.log(Math.trunc(false)); // 0
console.log(Math.trunc(NaN)); // NaN
console.log(Math.trunc(undefined)); // NaN
console.log(Math.trunc()); // NaN

// Math.sign()
// 参数为正数的时候，返回1
// 参数是负数的时候，返回-1
// 参数为0的时候，返回0
// 参数为-0的时候，返回-0
// 其他值的时候，返回NaN
console.log(Math.sign(3.14)) // 1
console.log(Math.sign(-3.14)) // -1
console.log(Math.sign(0)) // 0
console.log(Math.sign(NaN)) // NaN
console.log(Math.sign(true)) // 1
console.log(Math.sign(false)) // 0

console.log(Math.cbrt(3)); //27  立方
```
