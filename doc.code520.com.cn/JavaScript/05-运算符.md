### `运算符`

#### `运算符`

```js
// 运算符
// 算数运算符 自增/自减运算符 赋值运算符 比较运算符 逻辑运算符  条件运算符
// 算术运算符
// + - * / %(余)

// 自增/自减运算符
// ++  在原本值的基础上+1 -- 在原本值的基础上-1
var x = 1;
// ++x  x++
console.log(x++);// 1
console.log(x);// 2
console.log(++x);// 3
console.log(x);// 3
// ++写在后面，先赋值，再自增 先把原本的结果输出出来，然后再+1
//++在前，先自增，再赋值   先把原本的结果+1，然后再输出

// --和++用法一样

var y = 3;
console.log(y++);// 3
console.log(y--);// 4
console.log(y);// 3
console.log(y--);// 3
console.log(++y);// 3
console.log(--y);// 2

----

// 赋值运算符 = 赋值
// += -= *= /=
// +=在原本的值的基础上+xx
var x = 10;
var y = 20;
console.log(x+=y);// 30
// x = x+y;
// x += y;
console.log(x);// 30
```

#### `比较运算符`

```js
// 比较运算符
// ==  === !=  !== >   <   >=  <=

// 判断两个值是否相等 == ===
// ==不考虑数据类型  ===全等 ，数据类型也要相等
console.log(2 == '2');// t
console.log(2 === '2');// f
console.log(2 != '2');//  f
console.log(2 !== '2');// t

----

// 逻辑运算符
// 与或非
// &&与 要求前后两个表达式必须同时满足 true
// ||或 前后两个表达式满足一个就可以  true true/false
// !非 对当前的表达式进行否定 否定
// 条件表达式
var x = 1;
var y = "1";
var z = 2;
console.log((x==y)&&(x!=y));//false
console.log((x==y)||(x!=y));//true
console.log(!(x==y));//false

// !=      ===
// !==   不绝对等于
var x = "12";
var y = 12;
console.log(x!==y);// true
```

#### `三元运算符`

```js
// 条件运算符/三元运算符
// ?:
// 条件表达式?执行代码1:执行代码2
2>1?console.log("ture"):console.log("false");// true
```
