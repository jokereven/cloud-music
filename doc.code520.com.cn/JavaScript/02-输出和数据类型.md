### `输出和数据类型`

#### `输出方式`

```js
// console系列
// 控制台输出，目前用到最多的方式
// 通常用来检查报错

var txt = 'code520';
console.log(txt);

// 打印错误信息
console.error('error');

// 打印警告信息
console.warn('warn');

// 在页面写入内容
document.write('加载中...');

document.write('  code520   ');
// 浏览器的读取机制
// 无论在输出内容的时候写入了多少个空格，最后都显示成一个空格
document.write(
	'&nbsp;&nbsp;&nbsp;&nbsp;' + 'code520' + '&nbsp;&nbsp;&nbsp;&nbsp;'
);
document.write("<span style='white-space:pre;'>" + '  code520   ' + '</span>');
// 页面加载以后会执行的代码块
window.onload = function () {
	document.write('code520');
};
```

```js
// 两个值，用+相加，如果其中有一个值是字符串，那么就会作为一个连接的作用
// 如果多个值相加，从字符串开始 + 作为连接的作用

var num3 = 520;
var txt3 = 'code';
var num4 = 456;
console.log(num3 + num4 + txt3); // 976code
console.log(num3 + txt3 + num4); // 520code456
```
