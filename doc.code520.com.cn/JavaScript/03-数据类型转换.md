### `数据类型转换`

#### `数据类型转换`

```js
// 数据类型转换
// 字符串和数字
// 强制类型转换，非强制类型转换 显示转换/隐式转换
// 强制类型转换
// 字符串 -> 数字
// 1.parseInt() 取整,取整数部分
var txt = '520';
console.log(typeof txt); // string
console.log(txt + 10); // 530
var val = parseInt(txt);
console.log(val); //520
console.log(typeof val); // number
----
var txt = "3.14";
console.log(parseInt(txt)); // 3

var txt = "3.14pi";
console.log(parseInt(txt)); // 3
// 只负责转换，但是转换成啥，他不管
var txt = "pi";
console.log(parseInt(txt)); // NAN
console.log(typeof parseInt(txt)); // number
// NAN不是数字的数字类型
// 不是一个数字，但是他是数字类型
// 错误运算结果
console.log(undefined+3.14);// NAN
// NAN 不等任何值(包括本身)      我就是我，我就是这个色的烟火
// NAN不等于NAN
----
var txt1 = "+123";
var txt2 = "-123";
console.log(parseInt(txt1)); // 123
console.log(parseInt(txt2));// -123

// Math.abs()转换成正数
console.log(Math.abs(-3.14)); // 3.14

// parseFloat() 取浮点  说白了，除了整数部分保留以外，小数部分我也保留
// 保留第一个小数点以后的值，直到遇到了第二个小数点或者文字的时候停止
var txt = "123.456"; // 123.456
var txt = "123.456.789";// 123.456
var txt = "123.456文字.789";// 123.456
var txt = "文字123.456.789";// NAN
console.log(parseFloat(txt));
console.log(typeof parseFloat(txt));
----
// Number()普通转换
var txt = "456";// 456
var txt = '456code'; // NAN
console.log(Number(txt));
console.log(typeof Number(txt));

// parseInt()和Number()区别，都可以进行字符串到数字类型之间的转换，当转换的内容
// 包含非数字的时候，Number()就会返回NAN，parseInt()看情况，如果是数字开头，就
// 返回合法的数字部分，如果是文字开头，返回NAN
```

#### `复杂数据类型`

```js
// 万物皆对象
// 复杂数据类型: 数组，对象，函数;
// 键值对
// 对象 object
// 对象名字{
//     属性:值,
//     属性:值,
//     属性:值
// }
// 对象就是一系列属性和值的集合，每一对键值对之间用，来分隔开
var obj1 = {
    name:"joker",
    age:19
};
var obj2 = {
    name:"even",
    age:19,
};
// 获取属性值的方式
// []
console.log(obj1["name"]);
console.log(obj1["age"]);
console.log(obj1);

// .
console.log(obj1.name);
console.log(obj1.age);

----

// 如果当前的属性是数字，就需要用[]，而不是.
var obj1 = {
    00:"code",
    "01":520
}
// console.log(obj1.00);
console.log(obj1[00]);// code
console.log(obj1["00"]);// undefined

// 数组
// 记录每一个同学的身高
// 180  185 183 175
// 数组：一系列数据的集合
// arr是名字
var arr = [180, 185, 183, 175, 'NAN', { name: 'Xue', age: 36 }];
console.log(arr);
// 长度length .的
console.log(arr.length);
// 找数组当中的数据
// 数组当中每一条数据，都有自己对应的下标值 下标
// 180  0
// 185  1
// 183  2
// 150  3
// NAN 4
// 对象  5
// 下标值从0开始，长度从1开始
// 一个数组的最大下标值永远比长度小1

console.log(arr[4]); // NAN
```
