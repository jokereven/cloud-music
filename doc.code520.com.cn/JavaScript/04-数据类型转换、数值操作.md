### `数据类型转换、数值操作`

#### `隐式转换`

```js
// 隐式转换
// js弱类型脚本语言，语法不够严谨

// 显示转换/强制类型转换  字符串&数字
// +  隐式转换的一种  数字 -> 字符串
var txt = "code";
var num = 520;
console.log(typeof (txt+num));// string
// - * / %  字符串 -> 数字
----
var txt = "123";
var num = 456;
console.log(typeof(txt-num));// number -333
console.log(txt*num);// 56088
console.log(typeof(txt/num));// number
// %  取余    取余数   没有除尽的值
console.log(typeof(txt%num));// number
```

#### `数值操作`

```js
// Math.abs()绝对值
// 向上取整,如果有小数部分，整数部分+1
console.log(Math.ceil(3.14)); // 4
console.log(Math.ceil(3.0)); // 3
console.log(Math.floor(3.0)); // 3
// 向下取整,舍弃小数部分  和parseInt()类似
console.log(Math.floor(3.14)); // 3

// Math.max() 返回两者的较大值
console.log(Math.max(996, 007)); // 996
// 返回两者中的较小值
console.log(Math.min(996, 007)); // 007

// Math.random()  随机数  [0,1)
console.log(Math.random());
```
